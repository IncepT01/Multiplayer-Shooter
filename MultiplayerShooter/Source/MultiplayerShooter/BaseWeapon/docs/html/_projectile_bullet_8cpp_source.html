45 void AProjectileBullet::OnHit(UPrimitiveComponent* HitComp, AActor* OtherActor, UPrimitiveComponent* OtherComp, FVector NormalImpulse, const FHitResult& Hit)
46 {
47 AMainCharacter* OwnerCharacter = Cast<AMainCharacter>(GetOwner());
48 if (OwnerCharacter)
49 {
50 AMyPlayerController* OwnerController = Cast<AMyPlayerController>(OwnerCharacter->Controller);
51 if (OwnerController)
52 {
53 if (OwnerCharacter->HasAuthority() && !bUseServerSideRewind)
54 {
55 UE_LOG(LogTemp, Warning, TEXT("Server Damaging for: %f"), Damage * OwnerCharacter->GetDamageMultiplier());
56 UGameplayStatics::ApplyDamage(OtherActor, Damage * OwnerCharacter->GetDamageMultiplier(), OwnerController, this, UDamageType::StaticClass());
57 Super::OnHit(HitComp, OtherActor, OtherComp, NormalImpulse, Hit);
58 return;
59 }
60 AMainCharacter* HitCharacter = Cast<AMainCharacter>(OtherActor);
61 if (bUseServerSideRewind && OwnerCharacter->GetLagCompensation() && OwnerCharacter->IsLocallyControlled() && HitCharacter)
62 {
63 OwnerCharacter->GetLagCompensation()->ProjectileServerScoreRequest(
64 HitCharacter,
65 TraceStart,
66 InitialVelocity,
67 OwnerController->GetServerTime() - OwnerController->SingleTripTime
68 );
69 }
70 }
71 }
72
73 Super::OnHit(HitComp, OtherActor, OtherComp, NormalImpulse, Hit);
74 }
