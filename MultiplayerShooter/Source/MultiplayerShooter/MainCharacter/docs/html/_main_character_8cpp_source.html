443void AMainCharacter::AimOffset(float DeltaTime)
444{
445 if (Combat && Combat->EquippedWeapon == nullptr) return;
446 float Speed = CalculateSpeed();
447 bool bIsInAir = GetCharacterMovement()->IsFalling();
448
449 if (Speed == 0.f && !bIsInAir) // standing still, not jumping
450 {
451 bRotateRootBone = true;
452 FRotator CurrentAimRotation = FRotator(0.f, GetBaseAimRotation().Yaw, 0.f);
453 FRotator DeltaAimRotation = UKismetMathLibrary::NormalizedDeltaRotator(CurrentAimRotation, StartingAimRotation);
454 AO_Yaw = DeltaAimRotation.Yaw;
455 if (TurningInPlace == ETurningInPlace::ETIP_NotTurning)
456 {
457 InterpAO_Yaw = AO_Yaw;
458 }
459 bUseControllerRotationYaw = true;
460 TurnInPlace(DeltaTime);
461 }
462 if (Speed > 0.f || bIsInAir) // running, or jumping
463 {
464 bRotateRootBone = false;
465 StartingAimRotation = FRotator(0.f, GetBaseAimRotation().Yaw, 0.f);
466 AO_Yaw = 0.f;
467 bUseControllerRotationYaw = true;
468 TurningInPlace = ETurningInPlace::ETIP_NotTurning;
469 }
470
471 CalculateAO_Pitch();
472}

474void AMainCharacter::CalculateAO_Pitch()
475{
476 AO_Pitch = GetBaseAimRotation().Pitch;
477 if (AO_Pitch > 90.f && !IsLocallyControlled())
478 {
479 // map pitch from [270, 360) to [-90, 0)
480 FVector2D InRange(270.f, 360.f);
481 FVector2D OutRange(-90.f, 0.f);
482 AO_Pitch = FMath::GetMappedRangeValueClamped(InRange, OutRange, AO_Pitch);
483 }
484}

499void AMainCharacter::TurnInPlace(float DeltaTime)
500{
501 if (AO_Yaw > 90.f)
502 {
503 TurningInPlace = ETurningInPlace::ETIP_Right;
504 }
505 else if (AO_Yaw < -90.f)
506 {
507 TurningInPlace = ETurningInPlace::ETIP_Left;
508 }
509
510 if (TurningInPlace != ETurningInPlace::ETIP_NotTurning)
511 {
512 InterpAO_Yaw = FMath::FInterpTo(InterpAO_Yaw, 0.f, DeltaTime, 4.f);
513 AO_Yaw = InterpAO_Yaw;
514 if (FMath::Abs(AO_Yaw) < 15.f)
515 {
516 TurningInPlace = ETurningInPlace::ETIP_NotTurning;
517 StartingAimRotation = FRotator(0.f, GetBaseAimRotation().Yaw, 0.f);
518 }
519 }

553void AMainCharacter::ChatButtonPressed()
554{
555 MyPlayerController = MyPlayerController == nullptr
556 ? Cast<AMyPlayerController>(Controller)
557 : MyPlayerController;
558 if (MyPlayerController)
559 {
560 AMainHUD* MainHUD = Cast<AMainHUD>(MyPlayerController->GetHUD());
561
562 if (MainHUD)
563 {
564 // Assuming you already have a reference to your UMG widget, let's say ChatWidget is your widget
565 UChat* ChatWidget = MainHUD->Chat;
566 if (ChatWidget)
567 {
568 if (ChatWidget->InputField)
569 {
570 // Check if the input field has focus
571 if (ChatWidget->InputField->HasUserFocus(MyPlayerController))
572 {
573 UE_LOG(LogTemp, Display, TEXT("Chat Input was in focus -> removing focus"));
574
575 // Unfocus and re-enable player input
576 MyPlayerController->SetInputMode(FInputModeGameOnly());
577 MyPlayerController->bShowMouseCursor = false;
578 }
579 else
580 {
581 UE_LOG(LogTemp, Display, TEXT("Chat Input was not in focus -> focusing now"));
582
583 // Focus input field and disable movement/game input
584 ChatWidget->InputField->SetFocus();
585 MyPlayerController->SetInputMode(FInputModeUIOnly());
586 MyPlayerController->bShowMouseCursor = true;
587 }
588 }
589 }
590 else
591 {
592 UE_LOG(LogTemp, Display, TEXT("No ChatWidget"));
593 }
594 }
595 else
596 {
597 UE_LOG(LogTemp, Display, TEXT("No MainHUD"));
598 }
599 }
600 else
601 {
602 UE_LOG(LogTemp, Display, TEXT("No PlayerController"));
603 }
604 UE_LOG(LogTemp, Display, TEXT("ChatButtonPressed"));
605}

775void AMainCharacter::SimProxiesTurn()
776{
777 if (Combat == nullptr || Combat->EquippedWeapon == nullptr) return;
778 bRotateRootBone = false;
779 float Speed = CalculateSpeed();
780 if (Speed > 0.f)
781 {
782 TurningInPlace = ETurningInPlace::ETIP_NotTurning;
783 return;
784 }
785
786 ProxyRotationLastFrame = ProxyRotation;
787 ProxyRotation = GetActorRotation();
788 ProxyYaw = UKismetMathLibrary::NormalizedDeltaRotator(ProxyRotation, ProxyRotationLastFrame).Yaw;
789
790 if (FMath::Abs(ProxyYaw) > TurnThreshold)
791 {
792 if (ProxyYaw > TurnThreshold)
793 {
794 TurningInPlace = ETurningInPlace::ETIP_Right;
795 }
796 else if (ProxyYaw < -TurnThreshold)
797 {
798 TurningInPlace = ETurningInPlace::ETIP_Left;
799 }
800 else
801 {
802 TurningInPlace = ETurningInPlace::ETIP_NotTurning;
803 }
804 return;
805 }
806 TurningInPlace = ETurningInPlace::ETIP_NotTurning;
807}
