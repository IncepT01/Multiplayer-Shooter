64void AMainGameMode::Tick(float DeltaTime)
65{
66 Super::Tick(DeltaTime);
67
68 if (MatchState == MatchState::WaitingToStart)
69 {
70 CountdownTime = WarmupTime - GetWorld()->GetTimeSeconds() + LevelStartingTime;
71 if (CountdownTime <= 0.f)
72 {
73 StartMatch();
74 }
75 }
76 else if (MatchState == MatchState::InProgress)
77 {
78 CountdownTime = WarmupTime + MatchTime - GetWorld()->GetTimeSeconds() + LevelStartingTime;
79 if (CountdownTime <= 0.f)
80 {
81 SetMatchState(MatchState::Cooldown);
82 }
83 }
84 else if (MatchState == MatchState::Cooldown)
85 {
86 CountdownTime = CooldownTime + WarmupTime + MatchTime - GetWorld()->GetTimeSeconds() + LevelStartingTime;
87 if (CountdownTime <= 0.f)
88 {
89 RestartGame();
90 }
91 }
92}
