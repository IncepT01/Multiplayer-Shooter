238void AMyPlayerController::SetHUDTime()
239 {
240 float TimeLeft = 0.f;
241 if (MatchState == MatchState::WaitingToStart) TimeLeft = WarmupTime - GetServerTime() + LevelStartingTime;
242 else if (MatchState == MatchState::InProgress) TimeLeft = WarmupTime + MatchTime - GetServerTime() + LevelStartingTime;
243 else if (MatchState == MatchState::Cooldown) TimeLeft = CooldownTime + WarmupTime + MatchTime - GetServerTime() + LevelStartingTime;
244
245 uint32 SecondsLeft = FMath::CeilToInt(TimeLeft);
246 if (CountdownInt != SecondsLeft)
247 {
248 if (MatchState == MatchState::WaitingToStart || MatchState == MatchState::Cooldown)
249 {
250 SetHUDAnnouncementCountdown(TimeLeft);
251 }
252 if (MatchState == MatchState::InProgress)
253 {
254 SetHUDMatchCountdown(TimeLeft);
255 }
256 }
257
258 CountdownInt = SecondsLeft;
259 }

482void AMyPlayerController::HandleCooldown()
483{
484 UE_LOG(LogTemp, Warning, TEXT("End of match!"));
485 MainHUD = MainHUD == nullptr ? Cast<AMainHUD>(GetHUD()) : MainHUD;
486 if (MainHUD)
487 {
488 MainHUD->CharacterOverlay->RemoveFromParent();
489
490 if (!IsValid(MainHUD->Announcement))
491 {
492 MainHUD->AddAnnouncement();
493 }
494
495 bool bHUDValid = IsValid(MainHUD->Announcement) &&
496 IsValid(MainHUD->Announcement->AnnouncementText);
497
498 if (bHUDValid)
499 {
500 MainHUD->Announcement->SetVisibility(ESlateVisibility::Visible);
501 FString AnnouncementText("New Match Starts In:");
502 MainHUD->Announcement->AnnouncementText->SetText(FText::FromString(AnnouncementText));
503
504 SetHUDAnnouncementCountdown(CooldownTime);
505
506 AMainGameState* MainGameState = Cast<AMainGameState>(UGameplayStatics::GetGameState(this));
507 AMainPlayerState* MainPlayerState = GetPlayerState<AMainPlayerState>();
508 if (MainGameState && MainPlayerState)
509 {
510 TArray<AMainPlayerState*> TopPlayers = MainGameState->TopScoringPlayers;
511 FString InfoTextString;
512 if (TopPlayers.Num() == 0)
513 {
514 InfoTextString = FString("There is no winner.");
515 }
516 else if (TopPlayers.Num() == 1 && TopPlayers[0] == MainPlayerState)
517 {
518 InfoTextString = FString("You are the winner!");
519 }
520 else if (TopPlayers.Num() == 1)
521 {
522 InfoTextString = FString::Printf(TEXT("Winner: \n%s"), *TopPlayers[0]->GetPlayerName());
523 }
524 else if (TopPlayers.Num() > 1)
525 {
526 InfoTextString = FString("Players tied for the win:\n");
527 for (auto TiedPlayer : TopPlayers)
528 {
529 InfoTextString.Append(FString::Printf(TEXT("%s\n"), *TiedPlayer->GetPlayerName()));
530 }
531 }
532
533 MainHUD->Announcement->InfoText->SetText(FText::FromString(InfoTextString));
534 }
535 }
536 else
537 {
538 if (IsValid(MainHUD->Announcement))
539 {
540 if (IsValid(MainHUD->Announcement->AnnouncementText))
541 {
542 }
543 else
544 {
545 UE_LOG(LogTemp, Warning, TEXT("EoM: bHUDValid (AnnouncementText) is false!"));
546 }
547 }
548 else
549 {
550 UE_LOG(LogTemp, Warning, TEXT("EoM: bHUDValid (Announcement) is false!"));
551 }
552 }
553 }
554 else
555 {
556 UE_LOG(LogTemp, Warning, TEXT("EoM: MainHUD is null!"));
557 }
558
559 AMainCharacter* MainCharacter = Cast<AMainCharacter>(GetPawn());
560 if (MainCharacter && MainCharacter->GetCombat())
561 {
562 MainCharacter->bDisableGameplay = true;
563 if (MainCharacter->GetCombat()->GetFireButtonPressed())
564 {
565 MainCharacter->GetCombat()->FireButtonPressed(false);
566 }
567 }

590void AMyPlayerController::CheckPing(float DeltaTime)
591{
592 if (HasAuthority()) return;
593
594 HighPingRunningTime += DeltaTime;
595 if (HighPingRunningTime > CheckPingFrequency)
596 {
597 PlayerState = PlayerState == nullptr ? TObjectPtr<APlayerState>(GetPlayerState<APlayerState>()) : PlayerState;
598 if (PlayerState)
599 {
600 //UE_LOG(LogTemp, Warning, TEXT("PlayerState->ExactPing: %f"), PlayerState->ExactPing);
601 if (PlayerState->ExactPing > HighPingThreshold) // ping is compressed; it's actually ping / 4
602 {
603 HighPingWarning();
604 ServerReportPingStatus(true);
605 }
606 else
607 {
608 StopHighPingWarning();
609 ServerReportPingStatus(false);
610 }
611 }
612 HighPingRunningTime = 0.f;
613 }
