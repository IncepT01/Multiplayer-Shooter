146void UCombatComponent::EquipWeapon(ABaseWeapon* WeaponToEquip)
147{

241{
242 //UE_LOG(LogTemp, Warning, TEXT("Round spent in local fire"));
243 EquippedWeapon->Fire(fnHitTarget);
244 EquippedWeapon->SpendRound();
245 if (EquippedWeapon->MuzzleFlashComponent)
246 {
247 EquippedWeapon->MuzzleFlashComponent->Deactivate(); // Deactivate the Niagara system
248 EquippedWeapon->MuzzleFlashComponent->DestroyComponent(); // Destroy the component
249 EquippedWeapon->MuzzleFlashComponent = nullptr;

251 }
252 if (EquippedWeapon->MuzzleFlashNiagaraSystem && EquippedWeapon->WeaponMesh)
253 {
254 // Spawn the Niagara particle system at the muzzle socket and set it to loop
255 //UE_LOG(LogTemp, Warning, TEXT("MuzzleFlash spawning"));
256 EquippedWeapon->MuzzleFlashComponent = UNiagaraFunctionLibrary::SpawnSystemAttached(
257 EquippedWeapon->MuzzleFlashNiagaraSystem,
258 EquippedWeapon->WeaponMesh,
259 "Muzzle", // Socket name
260 FVector::ZeroVector,
261 FRotator::ZeroRotator,
262 EAttachLocation::SnapToTarget,
263 true
264 );
265 }
266
267 if (EquippedWeapon->FireSound)
268 {
269 UE_LOG(LogTemp, Warning, TEXT("Playing FireSound"));
270 UGameplayStatics::PlaySoundAtLocation(this, EquippedWeapon->FireSound, EquippedWeapon->GetActorLocation());

319void UCombatComponent::SetHUDCrosshairs(float DeltaTime)
320{
321 if (Character == nullptr || Character->Controller == nullptr) return;
322
323 Controller = Controller == nullptr ? Cast<AMyPlayerController>(Character->Controller) : Controller;
324 if (Controller)
325 {
326 HUD = HUD == nullptr ? Cast<AMainHUD>(Controller->GetHUD()) : HUD;
327 if (HUD)
328 {
329 if (EquippedWeapon)
330 {
331 HUDPackage.CrosshairsCenter = EquippedWeapon->CrosshairsCenter;
332 HUDPackage.CrosshairsLeft = EquippedWeapon->CrosshairsLeft;
333 HUDPackage.CrosshairsRight = EquippedWeapon->CrosshairsRight;
334 HUDPackage.CrosshairsBottom = EquippedWeapon->CrosshairsBottom;
335 HUDPackage.CrosshairsTop = EquippedWeapon->CrosshairsTop;
336 }
337 else
338 {
339 HUDPackage.CrosshairsCenter = nullptr;
340 HUDPackage.CrosshairsLeft = nullptr;
341 HUDPackage.CrosshairsRight = nullptr;
342 HUDPackage.CrosshairsBottom = nullptr;
343 HUDPackage.CrosshairsTop = nullptr;
344 }
345 //Spread calculation
346
347 // [0, 600] -> [0, 1]
348 FVector2D WalkSpeedRange(0.f, Character->GetCharacterMovement()->MaxWalkSpeed);
349 FVector2D VelocityMultiplierRange(0.f, 1.f);
350 FVector Velocity = Character->GetVelocity();
351 Velocity.Z = 0.f;
352
353 CrosshairVelocityFactor = FMath::GetMappedRangeValueClamped(WalkSpeedRange, VelocityMultiplierRange, Velocity.Size());
354
355 if (Character->GetCharacterMovement()->IsFalling())
356 {
357 CrosshairInAirFactor = FMath::FInterpTo(CrosshairInAirFactor, 2.25f, DeltaTime, 2.25f);
358 }
359 else
360 {
361 CrosshairInAirFactor = FMath::FInterpTo(CrosshairInAirFactor, 0.f, DeltaTime, 30.f);
362 }
363
364 if (bAiming)
365 {
366 CrosshairAimFactor = FMath::FInterpTo(CrosshairAimFactor, 0.58f, DeltaTime, 30.f);
367 }
368 else
369 {
370 CrosshairAimFactor = FMath::FInterpTo(CrosshairAimFactor, 0.f, DeltaTime, 30.f);
371 }
372
373 CrosshairShootingFactor = FMath::FInterpTo(CrosshairShootingFactor, 0.f, DeltaTime, 40.f);
374
375 HUDPackage.CrosshairSpread =
376 0.5f +
377 CrosshairVelocityFactor +
378 CrosshairInAirFactor -
379 CrosshairAimFactor +
380 CrosshairShootingFactor;
381
382 HUD->SetHUDPackage(HUDPackage);
383 }
384 }
